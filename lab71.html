The "break" keyword in loop control

virtual infinite loop: while(true) f(;;) = f(;true;)

Each iteration contains a conditional break

let i = 1, s = 0;
while (true) {
    s += i++; // (s = s + i, i = i + 1)
    if (i == 11) break;
}

The "continue" keyword in loop control

nice comparison: "break" = drop out college, but "continue" = leave but will come back

continue: quit the rest instruction in the current iteration, and go on to the next iteration

for(;;) {
    Task 1;
    if (T) {
        Task 2; 
    }
}
    
rewrite to
for(;;) {
    Task1;
    if (!T) continue;
    Task2;
}

Add all even numbers in [0, 100]

s = 0, i = 0;
for(;;) {
    if (i % 2 == 0)
        s += i;
    i++;
    if (i > 100) break;
}
    
for(;;) {
    if (i % 2 == 1) { i++; continue; }
    s += i;
    i++;
    if (i > 100) break;
}
