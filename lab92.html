function allprimes(n)
{
    let list = [];
    for (let k = 2; k < n; k++)
    {
        let isprime = true;
        for (let p of list)
        {
            if ( p >= Math.sqrt(k)) break;
            if (k % p === 0)
            {
                isprime = false;
                break;
            }
        }
        if (isprime)
        {
            console.log(k);
        }
    }
}
